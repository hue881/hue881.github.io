# ── Part A: Define a 5×5 grayscale image ───────────────────── image = [ [0, 0, 0, 0, 0], [0, 5, 5, 5, 0], [0, 5, 9, 5, 0], [0, 5, 5, 5, 0], [0, 0, 0, 0, 0], ] # ── Part B: Apply the Laplacian edge filter ─────────────────── filtered = [] for y in range(1, 4): # skip row 0 (no above) and row 4 (no below) new_row = [] for x in range(1, 4): # skip col 0 and col 4 center = image[y][x] up = image[y-1][x] # row above down = image[y+1][x] # row below left = image[y][x-1] # column left right = image[y][x+1] # column right value = center*4 - (up + down + left + right) new_row.append(value) filtered.append(new_row) # ── Part C: Print results ───────────────────────────────────── print("Filtered grid:", filtered) # ── Part C+: Label edges ────────────────────────────────────── for row in filtered: for v in row: if v > 10: print("EDGE!", v) else: print("flat", v) # Expected output: [[10, 15, 10], [15, 16, 15], [10, 15, 10]] # Corner pixels (10) → EDGE! Edge-center (15) → EDGE! Center (16) → EDGE!
